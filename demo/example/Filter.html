require(['Filter'], function (Filter) {

    var map = {
        '1': [2, 3, 5],
        '2': [1, 2, 4]
    };

    new Filter({
        prefix: 'ecl-ui-filter',
        main: $('.ecl-ui-filter')[0],
        groups: 'p',

        onChange: function (e) {

            if (e.key === 'type') {
                var value = e.value[0];

                if (map[value]) {
                    this.disableItems('special', map[value]);
                }
                else {
                    this.enableItems('special');
                }
            }
        }

    }).render();
});
